# ✅ ПРАВИЛЬНЫЙ NGINX КОНФИГ ДЛЯ traffic.onai.academy

Путь: /etc/nginx/sites-available/traffic.onai.academy

═══════════════════════════════════════════════════════════════

# HTTP → HTTPS redirect
server {
    listen 80;
    listen [::]:80;
    server_name traffic.onai.academy;
    return 301 https://$server_name$request_uri;
}

# HTTPS server для Traffic Dashboard
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name traffic.onai.academy;

    # ✅ КРИТИЧЕСКИ ВАЖНО: правильный root!
    root /var/www/traffic.onai.academy;
    index index.html;

    # SSL certificates
    ssl_certificate /etc/letsencrypt/live/traffic.onai.academy/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/traffic.onai.academy/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    # Security headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;

    # Gzip compression
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css text/xml text/javascript application/x-javascript application/xml+rss application/json application/javascript;

    # SPA routing - все запросы → index.html
    location / {
        try_files $uri $uri/ /index.html;
    }

    # Статика с кешем
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }

    # API проксирование на backend
    location /api/ {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_connect_timeout 600s;
        proxy_send_timeout 600s;
        proxy_read_timeout 600s;
    }

    # Логи
    access_log /var/log/nginx/traffic.onai.academy.access.log;
    error_log /var/log/nginx/traffic.onai.academy.error.log;
}
